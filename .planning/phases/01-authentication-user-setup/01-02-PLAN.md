Create **backend/src/auth/utils.js**:
```javascript
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const crypto = require('crypto');

const SALT_ROUNDS = 12; // per research: 12 rounds for production
const ACCESS_TOKEN_EXPIRY = '15m';
const REFRESH_TOKEN_EXPIRY = '7d';

async function hashPassword(plaintext) {
  return bcrypt.hash(plaintext, SALT_ROUNDS);
}

async function verifyPassword(plaintext, hash) {
  return bcrypt.compare(plaintext, hash);
}

function issueTokens(userId) {
  const accessToken = jwt.sign(
    { sub: userId, type: 'access' },
    process.env.JWT_SECRET,
    { expiresIn: ACCESS_TOKEN_EXPIRY }
  );
  const refreshToken = jwt.sign(
    { sub: userId, type: 'refresh' },
    process.env.JWT_REFRESH_SECRET,
    { expiresIn: REFRESH_TOKEN_EXPIRY }
  );
  return { accessToken, refreshToken };
}

function hashToken(rawToken) {
  return crypto.createHash('sha256').update(rawToken).digest('hex');
}

module.exports = { hashPassword, verifyPassword, issueTokens, hashToken };
```

async function findByEmail(email) {

async function findById(id) {
  const result = await query('SELECT * FROM users WHERE id = $1', [id]);
  return result.rows[0] || null;

  const result = await query(
    'INSERT INTO users (email, password_hash) VALUES ($1, $2) RETURNING *',

  );
  return result.rows[0];

  const result = await query(
    'UPDATE users SET password_hash = $1, updated_at = NOW() WHERE id = $2 RETURNING *',

  return result.rows[0];
}

  );
  return result.rows[0];

```javascript
const passport = require('passport');
const { Strategy: JwtStrategy } = require('passport-jwt');
const { findById } = require('../../users/model');

const cookieExtractor = (req) => {
  return req?.cookies?.accessToken || null;
};

passport.use(new JwtStrategy(
  {
    jwtFromRequest: cookieExtractor,
    secretOrKey: process.env.JWT_SECRET,
  },
  async (payload, done) => {
    try {
      if (payload.type !== 'access') return done(null, false);
      const user = await findById(payload.sub);
      if (!user) return done(null, false);
      return done(null, user);
    } catch (err) {
      return done(err, false);
    }
  }
));

module.exports = passport;
```

const passport = require('passport');
require('./strategies/jwt'); // Register strategy

function requireAuth(req, res, next) {
  passport.authenticate('jwt', { session: false }, (err, user) => {
    if (err) return next(err);
    if (!user) return res.status(401).json({ error: 'Authentication required' });
    req.user = user;
    next();
  })(req, res, next);
}

module.exports = { requireAuth };

Endpoint specs:

**POST /auth/signup** (rate-limited: 10/15min per IP):
- Body: { email, password, confirmPassword }
- Validate: email format, password >= 8 chars, confirmPassword matches
- Check email not already taken (return 409 if exists)
- Hash password with bcrypt 12 rounds
- Create user record
- Issue tokens, store refresh token hash in DB with expires_at = NOW() + 7 days
- Set accessToken cookie (maxAge: 15min) and refreshToken cookie (maxAge: 7 days)
- Return 201: { userId, email }
- Per D-12: no email verification, user is immediately logged in

**POST /auth/login** (rate-limited: 5/15min per IP):
- Body: { email, password }
- Find user by email; use generic error "Invalid email or password" for both not-found and wrong-password (anti-enumeration)
- Verify bcrypt hash; same generic error on failure
- Issue tokens, store refresh token hash in DB
- Set cookies
- Return 200: { userId, email }

**POST /auth/logout** (requires requireAuth):
- Read refreshToken from cookie
- If present: hash it and DELETE FROM refresh_tokens WHERE token_hash = $1
- Clear both cookies: res.clearCookie('accessToken', COOKIE_OPTS), res.clearCookie('refreshToken', COOKIE_OPTS)
- Return 200: { success: true }

**POST /auth/refresh** (no auth middleware — uses its own token check):
- Read refreshToken from cookie; 401 if missing
- jwt.verify(token, JWT_REFRESH_SECRET); 401 on failure
- Check token_hash exists in refresh_tokens table and expires_at > NOW(); 401 if not found
- DELETE old refresh_tokens record
- issueTokens(userId), store new refresh token hash
- Set new cookies
- Return 200: { success: true }

**POST /auth/forgot-password** (rate-limited: 3/hour per IP):
- Body: { email }
- Always return 200: { message: "If an account with that email exists, a reset link will be sent." } (anti-enumeration per pitfall 6)
- If user found: generate reset token with crypto.randomBytes(32).toString('hex'), hash it with SHA-256, store in password_reset_tokens with expires_at = NOW() + 1 hour
- Send email via Resend SDK:
  ```javascript
  const { Resend } = require('resend');
  const resend = new Resend(process.env.RESEND_API_KEY);
  await resend.emails.send({
    from: 'RoadTrip <noreply@roadtrip.app>',
    to: email,
    subject: 'Reset your RoadTrip password',
    html: `<p>Click <a href="${process.env.FRONTEND_URL}/reset-password?token=${resetToken}">here</a> to reset your password. Link expires in 1 hour.</p>`
  });
  ```

**POST /auth/reset-password** (rate-limited: 5/hour per IP):
- Body: { token, newPassword }
- Validate: newPassword >= 8 chars
- Hash token with SHA-256; query password_reset_tokens WHERE token_hash = $1 AND expires_at > NOW() AND used = FALSE
- 400 "Invalid or expired reset link" if not found
- Hash new password with bcrypt 12 rounds
- Update user password_hash
- Mark reset token used = TRUE
- Per D-07: issue new tokens, store refresh token, set cookies
- Return 200: { success: true, message: "Password reset. You are now logged in." }
