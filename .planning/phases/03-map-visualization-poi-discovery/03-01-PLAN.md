-- POIs table (Phase 3)
CREATE TABLE IF NOT EXISTS pois (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  stop_id UUID NOT NULL REFERENCES stops(id) ON DELETE CASCADE,
  osm_id BIGINT NOT NULL,
  osm_type VARCHAR(20) NOT NULL,
  name TEXT NOT NULL,
  category VARCHAR(100) NOT NULL,
  lat NUMERIC(9, 6) NOT NULL,
  lon NUMERIC(9, 6) NOT NULL,
  cuisine TEXT,
  opening_hours TEXT,
  website TEXT,
  phone TEXT,
  image_url TEXT,
  wikimedia_commons TEXT,
  cached_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_pois_stop_id ON pois(stop_id);
CREATE INDEX IF NOT EXISTS idx_pois_osm_id ON pois(osm_id, osm_type);

   - Import axios
   - `const OVERPASS_URL = 'https://overpass-api.de/api/interpreter'`
   - Define `SEARCH_TERM_MAP` object mapping user-friendly terms to Overpass tags:
     ```
     'restaurants': '["amenity"="restaurant"]',
     'coffee shops': '["amenity"="cafe"]',
     'cafes': '["amenity"="cafe"]',
     'bars': '["amenity"="bar"]',
     'hotels': '["tourism"~"hotel|motel"]',
     'camping': '["tourism"~"camp_site|caravan_site"]',
     'gas stations': '["amenity"="fuel"]',
     'supermarkets': '["shop"="supermarket"]',
     'hiking': '["route"="hiking"]',
     'museums': '["tourism"="museum"]',
     'parks': '["leisure"="park"]',
     'beaches': '["natural"="beach"]',
     'viewpoints': '["tourism"="viewpoint"]',
     'pharmacies': '["amenity"="pharmacy"]',
     'atm': '["amenity"="atm"]'
     ```
   - Export `SEARCH_TERM_MAP` (frontend needs the category list)
   - Export `POI_CATEGORIES` as `Object.keys(SEARCH_TERM_MAP)`
   - Export `async function queryPOIs(lat, lon, radius = 5000)`:
     - Default tags: `['["amenity"~"restaurant|cafe|bar|fast_food"]', '["tourism"~"attraction|museum|viewpoint|hotel|motel|camp_site|caravan_site"]', '["leisure"~"park|nature_reserve|beach_resort"]']`
     - Build Overpass QL query with `[out:json][timeout:25];` and `nwr{tag}(around:${radius},${lat},${lon});` for each tag, ending with `out center tags;`
     - POST to OVERPASS_URL with body `data=${encodeURIComponent(query)}` and header `Content-Type: application/x-www-form-urlencoded`
     - Map response.data.elements to POI objects: `{ osm_id: el.id, osm_type: el.type, lat: el.lat || el.center?.lat, lon: el.lon || el.center?.lon, name: el.tags?.name || 'Unnamed', category: el.tags?.amenity || el.tags?.tourism || el.tags?.leisure || 'other', cuisine: el.tags?.cuisine || null, opening_hours: el.tags?.opening_hours || null, website: el.tags?.website || null, phone: el.tags?.phone || null, image_url: el.tags?.image || null, wikimedia_commons: el.tags?.wikimedia_commons || null }`
     - Filter out results where both lat and lon are null
     - Return array of POI objects
   - Export `async function searchPOIs(lat, lon, searchTerm, radius = 5000)`:
     - Lowercase the searchTerm, look up in SEARCH_TERM_MAP
     - If found, use that tag; if not, try `["name"~"${searchTerm}",i]` as fallback
     - Build Overpass query with single tag, same format as queryPOIs
     - Return same POI object format

   - Import `{ query }` from '../db/connection'
   - Export `async function getCachedPOIs(stopId)`:

     - Return rows (empty array if cache expired or no results)
   - Export `async function cachePOIs(stopId, pois)`:
     - DELETE FROM pois WHERE stop_id = $1 (clear old cache for this stop)
     - For each poi, INSERT INTO pois (stop_id, osm_id, osm_type, name, category, lat, lon, cuisine, opening_hours, website, phone, image_url, wikimedia_commons) VALUES (...)
     - Use a single query with multi-row VALUES for efficiency: build parameterized query with $1=stopId and sequential params for each poi
   - Export `async function getPOIsByStopId(stopId)`:
     - SELECT * FROM pois WHERE stop_id = $1 ORDER BY category, name
     - Return rows
